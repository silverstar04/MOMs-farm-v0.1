rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isOwner(uid) {
      return signedIn() && request.auth.uid == uid;
    }

    function validSaveDocument(uid) {
      return request.resource.data.keys().hasOnly([
          "uid",
          "saveData",
          "updatedAt",
          "clientUpdatedAt",
          "saveVersion",
          "appVersion"
        ])
        && request.resource.data.uid == uid
        && request.resource.data.uid == request.auth.uid
        && request.resource.data.saveData is map
        && request.resource.data.updatedAt is timestamp
        && request.resource.data.clientUpdatedAt is number
        && request.resource.data.saveVersion is number
        && request.resource.data.appVersion is string;
    }

    function validRankingDocument() {
      return request.resource.data.keys().hasOnly([
          "playerId",
          "name",
          "score",
          "createdAt"
        ])
        && request.resource.data.playerId is string
        && request.resource.data.playerId.size() <= 80
        && request.resource.data.name is string
        && request.resource.data.name.size() >= 1
        && request.resource.data.name.size() <= 12
        && request.resource.data.score is number
        && request.resource.data.score >= 0
        && request.resource.data.createdAt is timestamp;
    }

    match /users/{uid}/save/main {
      allow read: if isOwner(uid);
      allow create, update: if isOwner(uid) && validSaveDocument(uid);
      allow delete: if false;
    }

    match /rankings/{rankingId} {
      allow read: if true;
      allow create, update: if signedIn() && validRankingDocument();
      allow delete: if false;
    }
  }
}
